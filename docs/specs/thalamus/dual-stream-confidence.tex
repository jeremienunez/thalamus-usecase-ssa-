\section{Context}
The Thalamus knowledge graph currently stores edges as untyped relations produced by the OSINT nano-swarm. In a dual-stream deployment (open-source intelligence fused with a high-trust field feed), edges produced from noisy open sources and edges produced from a tactical data-link, sensor-fusion bus, or mission debrief must be told apart at query time, at sweep time, and in any analyst-facing briefing. This spec defines the \textbf{confidence} scalar and \textbf{source\_class} discriminator carried on every edge, the promotion and demotion rules that move an edge between classes, and the invariants that prevent an OSINT-only claim from ever masquerading as a field-confirmed fact.

\section{Goals}
\begin{itemize}[leftmargin=1.2em]
  \item Every edge written to the knowledge graph carries a \texttt{confidence} $\in [0,1]$ and a \texttt{source\_class} $\in \{\texttt{FIELD\_HIGH}, \texttt{FIELD\_LOW}, \texttt{OSINT\_CORROBORATED}, \texttt{OSINT\_UNCORROBORATED}\}$.
  \item Promotion and demotion of an edge between classes is deterministic, auditable, and replayable from the edge history.
  \item Queries can filter and rank edges by source class and confidence without joining external tables.
  \item Sweep can target \texttt{OSINT\_UNCORROBORATED} edges older than a configurable window for analyst review.
  \item The edge schema is backward compatible: existing edges are migrated to \texttt{OSINT\_UNCORROBORATED} with \texttt{confidence = 0.3} unless a richer provenance is present.
\end{itemize}

\section{Non-Goals}
\begin{nongoal}
This spec does \emph{not} define the field-stream ingestion transport, the correlation algorithm, or the latency budget of the fusion loop. Those live in SPEC-TH-041 (Field-Correlation Cortex). It also does not define a probabilistic belief-propagation model across multi-hop paths --- confidence is a per-edge scalar, not a graph-wide Bayesian posterior.
\end{nongoal}

\section{Invariants}

\begin{invariant}
\textbf{I-1 --- Class floor.} An edge whose only provenance is an open-source observation is never written with \texttt{source\_class = FIELD\_HIGH} or \texttt{FIELD\_LOW}. OSINT provenance alone forbids a field class.
\end{invariant}

\begin{invariant}
\textbf{I-2 --- Promotion requires corroboration.} An OSINT edge is promoted to \texttt{OSINT\_CORROBORATED} only when at least one independent open source agrees within the corroboration window, \emph{or} a field event matches its entity pair and relation. Promotion to \texttt{FIELD\_HIGH} requires a field event \emph{and} a matching relation --- never a pure OSINT signal.
\end{invariant}

\begin{invariant}
\textbf{I-3 --- Field dominance.} A field event always overrides OSINT confidence on a matching edge. OSINT signals can never demote \texttt{FIELD\_HIGH}; only a contradicting field event, an explicit analyst override, or expiry of the field freshness window can.
\end{invariant}

\begin{invariant}
\textbf{I-4 --- Bounded confidence.} \texttt{confidence} is clamped to $[0, 1]$ at every write. Promotion and demotion apply monotonic transforms that respect the class ordering: \texttt{OSINT\_UNCORROBORATED} $\leq$ \texttt{OSINT\_CORROBORATED} $<$ \texttt{FIELD\_LOW} $\leq$ \texttt{FIELD\_HIGH} on the confidence axis within each class's allowed sub-range.
\end{invariant}

\begin{invariant}
\textbf{I-5 --- Auditability.} Every confidence or source-class change emits an \texttt{EdgeProvenanceEvent} with \texttt{actor} (\texttt{osint-swarm}, \texttt{field-correlation}, \texttt{analyst}, \texttt{sweep}), \texttt{reason}, \texttt{previous}, and \texttt{next} values. No silent mutations.
\end{invariant}

\section{Schema additions}

\begin{lstlisting}[language=TypeScript]
// Extends the ResearchEntityEdge row in packages/db-schema/src/knowledge-graph/edges.ts

export type SourceClass =
  | "FIELD_HIGH"              // confirmed by a high-trust field event
  | "FIELD_LOW"               // field event present but partial / low-rate
  | "OSINT_CORROBORATED"      // >= 2 independent OSINT sources agree
  | "OSINT_UNCORROBORATED";   // single OSINT hypothesis, not yet confirmed

export interface EdgeConfidence {
  value: number;              // [0, 1]
  sourceClass: SourceClass;
  lastPromotedAt: Date | null;
  lastDemotedAt: Date | null;
  corroborationCount: number; // distinct sources agreeing
  fieldEventId: string | null;// link to field event if class in {FIELD_*}
}

export interface EdgeProvenanceEvent {
  edgeId: number;
  actor: "osint-swarm" | "field-correlation" | "analyst" | "sweep";
  reason: string;
  previous: Pick<EdgeConfidence, "value" | "sourceClass">;
  next: Pick<EdgeConfidence, "value" | "sourceClass">;
  at: Date;
}

export interface PromoteEdgeInput {
  edgeId: number;
  evidence:
    | { kind: "osint-corroboration"; sources: string[] }
    | { kind: "field-match"; fieldEventId: string; stream: string };
}

export interface DemoteEdgeInput {
  edgeId: number;
  evidence:
    | { kind: "field-contradiction"; fieldEventId: string }
    | { kind: "analyst-override"; analystId: number; note: string }
    | { kind: "field-freshness-expired" };
}
\end{lstlisting}

\section{Promotion and demotion rules}

\begin{itemize}[leftmargin=1.2em]
  \item \textbf{OSINT swarm write} --- new edge always starts at \texttt{OSINT\_UNCORROBORATED}, \texttt{confidence = 0.30}, \texttt{corroborationCount = 1}.
  \item \textbf{OSINT corroboration} --- a second independent source for the same \texttt{(subject, relation, object)} within the corroboration window promotes to \texttt{OSINT\_CORROBORATED}, \texttt{confidence = 0.55}. Additional corroborating sources may raise \texttt{confidence} up to $0.75$ asymptotically.
  \item \textbf{Field match} --- a field event matching the edge promotes to \texttt{FIELD\_LOW} (partial field evidence) or \texttt{FIELD\_HIGH} (confirming field evidence on a critical stream), sets \texttt{fieldEventId}, \texttt{confidence} = the band's typical write-time value.
  \item \textbf{Field contradiction} --- a field event contradicting the edge demotes to \texttt{OSINT\_UNCORROBORATED}, \texttt{confidence = 0.15}, and flags the edge for analyst review via Sweep.
  \item \textbf{Field freshness expiry} --- a \texttt{FIELD\_*} edge whose backing field event is older than the freshness window degrades one class step, with \texttt{confidence} recomputed to the new band's upper bound.
  \item \textbf{Analyst override} --- manual set of \texttt{confidence} and \texttt{source\_class}, recorded as \texttt{analyst} actor. Cannot place an OSINT-only edge in a \texttt{FIELD\_*} class (enforced by I-1).
\end{itemize}

\section{Scenarios}

\begin{scenario}{OSINT edge starts uncorroborated}
\Given a cortex in the OSINT swarm emits a new edge for \texttt{(Actor A, associatedWith, Campaign C)} \\
\When \texttt{ConfidenceService.promote} is not yet called \\
\Then the edge is written with \texttt{source\_class = OSINT\_UNCORROBORATED} \\
\And \texttt{confidence = 0.30} \\
\And an \texttt{EdgeProvenanceEvent} is emitted with actor \texttt{osint-swarm}
\end{scenario}

\begin{scenario}{OSINT corroboration promotes confidence}
\Given the edge above exists at \texttt{confidence = 0.30} \\
\When a second independent OSINT source emits the same triple within the corroboration window \\
\Then the edge is promoted to \texttt{OSINT\_CORROBORATED} \\
\And \texttt{confidence = 0.55} \\
\And \texttt{corroborationCount = 2}
\end{scenario}

\begin{scenario}{Field match promotes to FIELD\_HIGH}
\Given an \texttt{OSINT\_CORROBORATED} edge at \texttt{confidence = 0.55} \\
\When a friendly-force tracking event matches the subject, relation, and object \\
\Then the edge is promoted to \texttt{FIELD\_HIGH} \\
\And \texttt{confidence = 0.92} \\
\And \texttt{fieldEventId} is populated \\
\And the provenance history records the transition
\end{scenario}

\begin{scenario}{Invariant I-1 forbids OSINT-only FIELD\_HIGH}
\Given an edge with only OSINT provenance \\
\When a caller attempts \texttt{promote} with no \texttt{field-match} evidence \\
\Then the call is rejected with \texttt{InvalidPromotion} \\
\And the edge class and confidence are unchanged \\
\And no \texttt{EdgeProvenanceEvent} is emitted
\end{scenario}

\begin{scenario}{Field contradiction demotes to OSINT\_UNCORROBORATED}
\Given a \texttt{FIELD\_LOW} edge at \texttt{confidence = 0.75} \\
\When a contradicting field event arrives on the tactical data-link \\
\Then the edge is demoted to \texttt{OSINT\_UNCORROBORATED} \\
\And \texttt{confidence = 0.15} \\
\And the edge is flagged for Sweep analyst review
\end{scenario}

\begin{scenario}{Field freshness expiry}
\Given a \texttt{FIELD\_HIGH} edge whose backing field event is older than the freshness window \\
\When the freshness sweep runs \\
\Then the edge is degraded one class step to \texttt{FIELD\_LOW} \\
\And \texttt{confidence} is recomputed to the upper bound of the \texttt{FIELD\_LOW} band \\
\And the provenance event records \texttt{field-freshness-expired}
\end{scenario}

\section{Acceptance Criteria}

\begin{ac}{AC-1}
Every edge insert produced by any OSINT cortex persists with \texttt{source\_class = OSINT\_UNCORROBORATED} and \texttt{confidence} in $[0.10, 0.50]$.
\end{ac}

\begin{ac}{AC-2}
A second independent OSINT source on an existing edge promotes it to \texttt{OSINT\_CORROBORATED} and records an \texttt{EdgeProvenanceEvent}.
\end{ac}

\begin{ac}{AC-3}
\texttt{ConfidenceService.promote} with \texttt{field-match} evidence moves the edge to \texttt{FIELD\_HIGH} (or \texttt{FIELD\_LOW} per stream policy) and sets \texttt{fieldEventId}.
\end{ac}

\begin{ac}{AC-4}
\texttt{ConfidenceService.promote} without \texttt{field-match} evidence never produces a \texttt{FIELD\_*} class. The call fails with a typed error.
\end{ac}

\begin{ac}{AC-5}
A contradicting field event demotes a \texttt{FIELD\_*} edge to \texttt{OSINT\_UNCORROBORATED} with \texttt{confidence} $\leq 0.20$.
\end{ac}

\begin{ac}{AC-6}
No OSINT actor can demote a \texttt{FIELD\_HIGH} edge. Attempts are rejected and logged.
\end{ac}

\begin{ac}{AC-7}
\texttt{ConfidenceService.history(edgeId)} returns every class or confidence transition in insertion order. No gaps.
\end{ac}

\begin{ac}{AC-8}
\texttt{ConfidenceService.query} can return all \texttt{OSINT\_UNCORROBORATED} edges older than a caller-supplied timestamp, for Sweep to pick up.
\end{ac}

\section{Open Questions}
\begin{itemize}[leftmargin=1.2em]
  \item Should \texttt{FIELD\_LOW} vs \texttt{FIELD\_HIGH} be a property of the stream, of the event, or of the correlation score? Current draft: stream policy sets the default, per-event override allowed.
  \item Freshness window per stream: a single global TTL, or per-stream configurable? Leaning per-stream.
  \item Do we persist provenance events in Postgres (auditable SQL) or ship them to the observability bus only? Current draft: both --- SQL for audit, bus for dashboards.
  \item Multi-hop confidence: out of scope for this spec but should be tracked in a follow-up.
\end{itemize}

\section{Context}
The knowledge base grows faster than a human can audit it. We need a continuous, cost-bounded process that scans grouped records, flags inconsistencies (missing fields, suspicious values, broken relationships), and hands the output to a human reviewer. This spec defines the \emph{producer} side: a batched swarm of small language-model calls that emits draft findings into a pending buffer. It is never allowed to mutate the source of truth.

\section{Goals}
\begin{itemize}[leftmargin=1.2em]
  \item Scan the full entity space in bounded wall-time and bounded token cost per run.
  \item Partition records into stable batches so each nano call sees a coherent, small context.
  \item Emit structured findings (category, severity, affected count, optional resolution payload) into the pending buffer.
  \item Incorporate past reviewer feedback (accepted / rejected) into subsequent prompts to reduce repeat false-positives.
  \item Surface a machine-readable \texttt{SweepResult} to trigger downstream notification.
\end{itemize}

\section{Non-Goals}
\begin{nongoal}
This service does \emph{not} write to the source-of-truth database, does not decide whether a finding is correct, and does not notify end users directly. Acceptance, rate-limiting, and dispatch are the responsibility of SPEC-SW-002; application of accepted fixes is the responsibility of SPEC-SW-003.
\end{nongoal}

\section{Invariants}
\begin{invariant}
The sweep service writes only to the pending finding buffer (Redis). It MUST NOT issue \texttt{INSERT} / \texttt{UPDATE} / \texttt{DELETE} against the source-of-truth tables.
\end{invariant}
\begin{invariant}
Every emitted finding has a bounded shape: \texttt{category} $\in$ known enum, \texttt{severity} $\in$ \{\texttt{critical}, \texttt{warning}, \texttt{info}\}, \texttt{title} $\le$ 200 chars, \texttt{description} $\le$ 1000 chars. Unknown values are coerced to safe defaults; malformed JSON yields an empty batch result, never a crash.
\end{invariant}
\begin{invariant}
A single sweep run produces a monotonically non-decreasing counter: \texttt{totalCalls} $\ge$ \texttt{successCalls} $\ge 0$, and \texttt{suggestionsStored} $\ge 0$.
\end{invariant}

\section{Interface}

\begin{lstlisting}[language=TypeScript]
export interface SweepResult {
  totalGroups: number;       // partitions scanned
  totalCalls: number;        // nano calls attempted
  successCalls: number;      // calls that returned valid JSON
  suggestionsStored: number; // findings written to pending buffer
  wallTimeMs: number;
  estimatedCost: number;     // USD, computed from token accounting
}

export type SweepCompleteCallback = (r: SweepResult) => Promise<void>;

export class NanoSweepService {
  constructor(entityRepo: EntityRepository, findingRepo: FindingRepository);
  onComplete(cb: SweepCompleteCallback): void;
  sweep(maxGroups?: number, mode?: SweepMode): Promise<SweepResult>;
}
\end{lstlisting}

\section{Scenarios}

\begin{scenario}{Nominal batch scan}
\Given the entity repository returns $N$ groups with aggregated stats \\
\When \texttt{sweep()} is invoked with no cap \\
\Then the service builds $\lceil N / \text{BATCH\_SIZE} \rceil$ batches \\
\And dispatches them through the nano wave caller \\
\And every parsed finding is appended to the pending buffer \\
\And onComplete callbacks fire exactly once with the aggregate \texttt{SweepResult}.
\end{scenario}

\begin{scenario}{Partial nano failure}
\Given 10 batches are dispatched \\
\When 3 nano calls return non-parseable text or error out \\
\Then \texttt{successCalls = 7} and \texttt{totalCalls = 10} \\
\And the 7 successful batches still produce findings \\
\And the run completes without throwing.
\end{scenario}

\begin{scenario}{Feedback-informed prompt}
\Given the feedback log contains a prior rejection for category \texttt{price\_anomaly} on group \emph{G} \\
\When a new batch contains group \emph{G} \\
\Then the prompt sent to nano includes that prior rejection as a bullet \\
\And the parser coerces any returned category outside the enum to \texttt{enrichment}.
\end{scenario}

\begin{scenario}{Cap respected}
\Given the repository returns 1000 groups \\
\When \texttt{sweep(maxGroups=50)} is called \\
\Then exactly 50 groups are scanned \\
\And the remaining 950 are not read, not prompted, not charged.
\end{scenario}

\begin{scenario}{No writes to source-of-truth}
\Given the sweep runs to completion \\
\When the source-of-truth tables are inspected before and after \\
\Then their row counts and contents are byte-identical.
\end{scenario}

\section{Acceptance Criteria}

\begin{ac}{AC-1}
Given $N$ groups and a batch size $B$, the service issues exactly $\lceil N / B \rceil$ nano calls.
\end{ac}
\begin{ac}{AC-2}
When a nano response is not valid JSON, the batch contributes zero findings and the run does not throw.
\end{ac}
\begin{ac}{AC-3}
Every finding written to the pending buffer has \texttt{category}, \texttt{severity}, \texttt{title}, \texttt{description} satisfying the bounded-shape invariant.
\end{ac}
\begin{ac}{AC-4}
No \texttt{INSERT} / \texttt{UPDATE} / \texttt{DELETE} is issued against source-of-truth tables during a sweep (verified by DB hook / spy).
\end{ac}
\begin{ac}{AC-5}
\texttt{onComplete} callbacks are invoked exactly once per sweep, after the pending buffer write has committed.
\end{ac}
\begin{ac}{AC-6}
The \texttt{estimatedCost} field is non-negative and reproducible for identical input/output token counts.
\end{ac}
\begin{ac}{AC-7}
When \texttt{maxGroups} is provided, the repository is queried for at most that many groups and no more nano calls are issued than $\lceil \text{maxGroups} / B \rceil$.
\end{ac}

\section{Open Questions}
\begin{itemize}[leftmargin=1.2em]
  \item Should batch size adapt to group cardinality, or stay a tuned constant?
  \item Do we persist raw nano responses for audit, or only parsed findings?
  \item How is partial-run resumability handled on worker crash?
\end{itemize}

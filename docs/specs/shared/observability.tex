\section{Context}
Every package in the monorepo needs a uniform way to emit structured logs and numeric metrics, so that operators can correlate behaviour across services with a single dashboard and alerting stack. The \texttt{observability} module in \texttt{@interview/shared} exposes two primitives: \texttt{createLogger(serviceName)} (Pino-based, environment-aware) and \texttt{MetricsCollector} (Prometheus registry with counters and histograms). This spec fixes their public contract so that services depend on stable behaviour rather than on transport implementation details.

\section{Goals}
\begin{itemize}[leftmargin=1.2em]
  \item Provide one logger factory whose output format adapts to the environment (pretty + file in dev, stdout and optional Loki in prod, silent in test) without changing caller code.
  \item Provide a metrics collector that owns its own \texttt{Registry}, tags all series with \texttt{app} and \texttt{env}, and exposes a single \texttt{getMetrics()} for the Prometheus scrape endpoint.
  \item Never leak secrets: authorization headers and cookies are redacted at the logger boundary.
  \item Degrade safely: a missing \texttt{LOKI\_HOST} or a transport error does not take the process down.
\end{itemize}

\section{Non-Goals}
\begin{nongoal}
This spec does not define per-service log levels, per-service metric names, or alert rules. Those are owned by the service that instantiates the primitives. This spec also does not cover distributed tracing (OpenTelemetry spans), which is a separate concern.
\end{nongoal}

\section{Invariants}
\begin{invariant}
Every log record produced by \texttt{createLogger(name)} carries at minimum the base fields \texttt{service = name} and \texttt{env = process.env.NODE\_ENV || "development"}.
\end{invariant}

\begin{invariant}
No log record emitted by \texttt{createLogger} contains the keys \texttt{req.headers.authorization} or \texttt{req.headers.cookie}, regardless of caller input, because Pino \texttt{redact.remove} is configured unconditionally.
\end{invariant}

\begin{invariant}
Every metric registered through a \texttt{MetricsCollector} instance is attached to that instance's \texttt{Registry}, and every series carries the default labels \texttt{app} and \texttt{env}. No metric leaks into the global Prometheus registry.
\end{invariant}

\begin{invariant}
When \texttt{NODE\_ENV === "test"}, \texttt{createLogger} returns a silent logger (level \texttt{"silent"}) so that test output is not polluted.
\end{invariant}

\section{Interface}

\begin{lstlisting}[language=TypeScript]
import type { Logger } from "pino";
import { Registry, Counter, Histogram } from "prom-client";

export function createLogger(serviceName: string): Logger;
export type { Logger };

export interface MetricsConfig {
  serviceName: string;
  enableDefaultMetrics?: boolean; // default true
}

export class MetricsCollector {
  readonly registry: Registry;
  constructor(config: MetricsConfig);

  createCounter(
    name: string,
    help: string,
    labelNames?: string[]
  ): Counter<string>;

  createHistogram(
    name: string,
    help: string,
    buckets?: number[],
    labelNames?: string[]
  ): Histogram<string>;

  getMetrics(): Promise<string>; // Prometheus text format
}
\end{lstlisting}

\section{Scenarios}

\begin{scenario}{Development logger wires pretty + file transports}
\Given \texttt{NODE\_ENV = "development"} and \texttt{LOG\_LEVEL} unset \\
\When \texttt{createLogger("api")} is called \\
\Then the returned logger has level \texttt{"debug"} \\
\And emits to \texttt{pino-pretty} and to \texttt{./logs/server.log} \\
\And base fields include \texttt{\{ service: "api", env: "development" \}}.
\end{scenario}

\begin{scenario}{Test logger is silent}
\Given \texttt{NODE\_ENV = "test"} \\
\When \texttt{createLogger("anything")} is called \\
\Then the returned logger has level \texttt{"silent"} \\
\And emits no records for \texttt{info}, \texttt{warn}, or \texttt{error} calls.
\end{scenario}

\begin{scenario}{Production logger without Loki}
\Given \texttt{NODE\_ENV = "production"} and \texttt{LOKI\_HOST} unset \\
\When \texttt{createLogger("api")} is called \\
\Then the logger writes to stdout (\texttt{destination: 1}) at level \texttt{info} \\
\And does not attempt to open a Loki connection.
\end{scenario}

\begin{scenario}{Production logger with Loki}
\Given \texttt{NODE\_ENV = "production"} and \texttt{LOKI\_HOST = "http://loki:3100"} \\
\When \texttt{createLogger("api")} is called \\
\Then the logger fans out to stdout and to \texttt{pino-loki} with labels \texttt{\{ app: "api", env: "production" \}} \\
\And transport errors are silenced (\texttt{silenceErrors: true}) so they do not crash the process.
\end{scenario}

\begin{scenario}{Secrets are redacted}
\Given any logger produced by \texttt{createLogger} \\
\When a record is logged with \texttt{req.headers.authorization = "Bearer xyz"} and \texttt{req.headers.cookie = "sid=..."} \\
\Then the emitted record contains neither key (they are \emph{removed}, not replaced by a marker).
\end{scenario}

\begin{scenario}{Metrics are isolated per collector}
\Given two \texttt{MetricsCollector} instances \texttt{A("svc-a")} and \texttt{B("svc-b")} \\
\When each registers a counter \texttt{requests\_total} \\
\Then each counter lives only in its own \texttt{registry} \\
\And \texttt{A.getMetrics()} returns series tagged \texttt{app="svc-a"} only \\
\And the global \texttt{prom-client} default registry stays untouched.
\end{scenario}

\section{Acceptance Criteria}

\begin{ac}{AC-1}
\texttt{createLogger(name)} returns a Pino logger whose \texttt{bindings()} include \texttt{service === name} and \texttt{env === process.env.NODE\_ENV || "development"}.
\end{ac}

\begin{ac}{AC-2}
In \texttt{NODE\_ENV = "test"}, the logger's level is \texttt{"silent"} and no record is written to any transport.
\end{ac}

\begin{ac}{AC-3}
In \texttt{NODE\_ENV = "development"}, the logger emits both to a pretty console transport and to \texttt{./logs/server.log}; the log level defaults to \texttt{"debug"} and respects \texttt{LOG\_LEVEL} when set.
\end{ac}

\begin{ac}{AC-4}
In \texttt{NODE\_ENV = "production"}, the logger writes to stdout; if and only if \texttt{LOKI\_HOST} is set, a \texttt{pino-loki} transport is added with labels \texttt{\{ app, env \}} and \texttt{silenceErrors: true}.
\end{ac}

\begin{ac}{AC-5}
Records containing \texttt{req.headers.authorization} or \texttt{req.headers.cookie} are emitted with those paths \emph{removed}, for every environment.
\end{ac}

\begin{ac}{AC-6}
\texttt{new MetricsCollector(\{ serviceName \})} constructs a fresh \texttt{Registry} and sets default labels \texttt{\{ app: serviceName, env \}}; \texttt{enableDefaultMetrics !== false} also registers \texttt{prom-client} default metrics under the \texttt{app\_} prefix.
\end{ac}

\begin{ac}{AC-7}
\texttt{createCounter} and \texttt{createHistogram} return prom-client instances that are registered only on the owning collector's \texttt{registry}, never on the global default registry.
\end{ac}

\begin{ac}{AC-8}
\texttt{getMetrics()} returns the Prometheus text exposition format string produced by the owning registry, including every metric created via \texttt{createCounter} / \texttt{createHistogram}.
\end{ac}

\section{Open Questions}
\begin{itemize}[leftmargin=1.2em]
  \item Should the redaction list be extensible per caller (e.g.\ additional PII paths) or is the fixed set sufficient?
  \item Is the \texttt{./logs/server.log} path configurable, or should dev builds always rely on \texttt{fly} / container log capture in line with production?
  \item Should \texttt{MetricsCollector} expose a \texttt{contentType} alongside \texttt{getMetrics()} so HTTP handlers set the correct Prometheus \texttt{Content-Type} without importing \texttt{prom-client} directly?
\end{itemize}
